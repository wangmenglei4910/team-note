函数的返回值

     function add(a,b){
        // reuturn 后面指定的是什么值 返回的就是什么值
        // 写了return 没有指定返回值 也是undefined
        // 没有指定返回值时 默认返回undefined
        return a+b
    }
    /*
        函数名()代表的意思：
            1，函数执行 
            2，这个整体也代表函数执行后的返回值
    */ 

    // 传入两个实参 进行求和并且接受函数返回的结果
    var sum = add(10,30)
    console.log(sum)

arguments

    /*
        函数中接收参数的两种方式：
            1，通过形参接收
            2，arguments接收

        形参和arguments有一种映射关系
        形参的值变化 arguments中的对应值也会跟着改变
        arguments 的值变了 对应的形参值也会跟着变化
    */
    function fe() {
        // arguments 类数组 实参集合 传入多少个实参 argument就有多少个实参
        // arguments.length 代表当前传入实参的个数
        // console.log(arguments)
        for (var i = 0; i < arguments.length; i++) {
            console.log(arguments[i])
        }
    }
    fe(10, 20, 30, 40, 50)

事件函数

元素.onclick = function(){}
事件函数中的this指的是当前绑定实践的元素

    // 根据点击的行为 做相应的事情 事件
    var box = document.getElementById('box');
    box.onclick = function(){
        // this指的是当前绑定事件的元素
        alert('点击了box')
        box.style.backgroundColor = 'lightblue'
        box.style.width = '500px'
    }
    console.dir(box);
