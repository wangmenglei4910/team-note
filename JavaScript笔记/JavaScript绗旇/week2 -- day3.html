字符串

charAt()
  • 作用：根据字符查找索引
  • 参数：指定字符
  • 返回值：指定字符对应的索引
  • 不会改变原字符串
    var strCharAt = "abcde"
    console.log(strCharAt.charAt(2)) // => c
    // str.charAt()和str[]两种方式查找不存在的字符
    console.log(strCharAt.charAt(20)) // => ""
    console.log(strCharAt[20]) // => undefined
charCodeAt()
  • 作用：从字符串中根据字符的索引位置 返回字符的unicode编码
  • 参数：索引值
  • 返回值：指定字符的unicode编码
  • 不会改变原字符串
    var strcharCodeAt = '1'
    console.log(strcharCodeAt.charCodeAt(0)) // => 49
fromCharCode()
  • 作用：根据unicode编码 返回字符
  • 参数：unicode编码
  • 返回值：字符
  • 不会改变原字符
    console.log(String.fromCharCode(49)) // => 1
concat()
  • 作用：拼接字符
  • 参数：任意字符
  • 返回值：拼接后的字符
  • 不会改变原字符串
    var strConcat = 'Java'
    var strConcat2 = 'Script'
    var res = strConcat.concat(strConcat2, "5.0")
    console.log(res) // => "JavaScript5.0"
    console.log(strConcat + strConcat2 + "5.0") // => "JavaScript5.0"
    console.log(`${strConcat}${strConcat2}5.0`) // => "JavaScript5.0"
indexof(x,[fromIndex])
  • 作用：根据字符查找索引位置
  • 参数；字符，字符查询的起始位置
  • 返回值：索引值 查找不到返回 -1
  • 不会改变原字符串
     var strIndexof = 'indexof'
    console.log(strIndexof.indexOf('q')) // => -1 查找不到返回-1
lastIndexof(x,[fromIndex])
  • 作用：根据字符查找索引位置
  • 参数：字符 开始查询的起始位置
  • 返回值：字符在字符串中的索引值 查找不到返回-1
  • 不会改变原字符串
  var strLastIndexof = 'lastindexof'
  console.log(strLastIndexof.lastIndexOf('q')) // => 1
match()
  • 作用：在字符串内查找指定的值
  • 参数：指定字符或者正则表达式
  • 返回值：根据指定字符查找到的一个或多个
  • 不会改变原字符串
    var strMatch = 'matchh';
    console.log(strMatch.match('h')) // => ["h", index: 4, input: "match", groups: undefined]
    // 利用正则将字符串中符合条件的查找出来 放进新的数组里面
    console.log(strMatch.match(/h/g)) // =>  ["h", "h"]

    var strMatch2 = 'hello 2019 byebye 2018'
    console.log(strMatch2.match(/\d+/g)) // ["2019", "2018"]
replace(x,y)
  • 作用：替换 默认情况下只能替换一次
  • 参数：x:要替换的那一项 y：要替换的值
  • 返回值：替换好的新字符串
  • 不会改变原字符串
    var strRep = 'hello world'
    var resR = strRep.replace('h', 'ooo')
    console.log(strRep)
    console.log(resR) // => "oooello world"
    console.log(strRep.replace(/o/g, '9')) // => "hell9 w9rld"
slice(x,y) 支持负数
  • 作用：字符串截取
  • 参数：x：起始索引 y:结束索引
  • 返回值：截取的字符串 返回的新的字符串
  • 不会改变原字符串
    var strSlice = 'hello world'
    console.log(strSlice.slice(1, 3)) // => el
substr(x,n)
  • 作用：从索引x开始截取 截取n个
  • 参数：x：索引位置 n:要截取的个数
  • 返回值：截取后的新的字符串
  • 不会改变原字符串
    var strSubstr = 'hello world'
    console.log(strSubstr.substr(1, 5)) // => 'ello '  //加上空格
subString(x,y)
  • 作用：从索引x截取到索引y
  • 参数：x：截取起始索引 y：截取结束索引
  • 返回值：截取后的新的字符串
  • 不会改变原字符串
    var strSubstring = 'hello world'
    console.log(strSubstring.substring(1, 7)) // => ello w
split()
  • 作用：根据指定的字符拆分成数组
  • 参数：指定的字符
  • 返回值：被分割符分割后组成的新数组
  • 不会改变原字符串
       var strSplit = 'hello world'

    var resSplit1 = strSplit.split()
    console.log(resSplit1) // => ["hello world"]

    var resSplit2 = strSplit.split(" ")
    console.log(resSplit2) // => ["hello", "world"]

    var resSplit3 = strSplit.split("")
    console.log(resSplit3) // => ["h", "e", "l", "l", "o", " ", "w", "o", "r", "l", "d"]

    var resSplit4 = strSplit.split("o")
    console.log(resSplit4) // => ["hell", " w", "rld"]

    var strSplit2 = 'a1b2c3d'

    var resSplit5 = strSplit2.split(/\d+/g)
    console.log(resSplit5) // => ["a", "b", "c", "d"]

    var strSplit3 = 'hello world'
    var arr = strSplit3.split(' ').join(" ")

toUpperCase()
  • 作用：将字符串的每项字符转换成大写的
  • 参数：无
  • 返回值：每一项都是大写的新的字符串
  • 不会改变原字符串
    var strUp = 'hello world'
    var resUp = strUp.toLocaleUpperCase();
    console.log(resUp) // "HELLO WORLD"
toLocalCase()
  • 作用：将字符串的每项字符转换成小写
  • 参数：无
  • 返回值：每一项都变成小写的新的字符串
  • 不会改变原字符串
    var strLow = resUp.toLowerCase();
    console.log(strLow) // => "hello world"
trim()
  • 作用：去除字符串两边的空格
  • 参数：无
  • 返回值：去除两边空格的新的字符串
  • 不会改变原字符串
     var strTirm = ' hello world '
    var resTirm = strTirm.trim();
    console.log(resTirm) // => "hello world"
includes()
  • 作用：判断一个字符是否包含在字符串中
  • 参数：指定字符
  • 返回值：true 包含 false 不包含
    var strInCludes = 'hello world'
    var resIn = strIndexof.includes('o')
    console.log(resIn) // true

   // 获取节点的父节点
