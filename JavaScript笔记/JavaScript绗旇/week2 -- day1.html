数组的常用方法

数组

1,定义数组：

  • var 数组名 = [值1,值2,...]
  • 用typeof检测出来的数组是 'object'

2,数组的特性

  • 索引 便于我们通过索引获取其中的某一项
  • 从0开始排列 数组中每一项都有对应的索引值
  • 通过标签名获取到的元素集合 称之为类数组
  • 函数中实参集合arguments 也称之为类数组

数组的方法

push

1，向数组的末尾添加一项或多项
2，参数：任意数据类型的值
3，返回值：添加后的数组的长度
4，原数组会发生改变

    var arrPush = [1,3,'hello']
    var dataPush = arrPush.push('4','world') 
    console.log(arrPush) //[1, 3, "hello", "4", "world"]
    console.log(dataPush) //返回值是添加后的数组的长度

unshift

1，向数组的开头添加一项或多项
2，参数：任意数据类型的值
3，返回值：添加后的数组的长度
4，原数组会发生改变

    var arrUnShift = [2,3,4]
    var dataUnShift = arrUnShift.unshift(1);
    console.log(arrUnShift) // [1, 2, 3, 4]
    console.log(dataUnShift) //返回值是添加后的数组的长度

pop

1，删除数组的最后一项
2，参数：无
3，返回值：删除掉的那一项
4，原数组会发生改变

    var arrPop = [1,2,3,4,'hello']
    var dataPop = arrPop.pop()
    console.log(arrPop) //  [1, 2, 3, 4]
    console.log(dataPop) //返回值是删除的那一项 'hello'

shift

1，删除数组的第一项
2，参数：无
3，返回值：删除的那一项
4，原数组会发生改变

    var arrShift = [0,1,2,3]
    var dataShift = arrShift.shift();
    console.log(arrShift) // [1, 2, 3]
    console.log(dataShift) // 返回值是删除的那一项 0

splice

1，添加、删除数组中的元素
2，参数：

  • splice(x) 从索引x处开始 一直删除到末尾，将删除的内容组成新数组返回
  • splice(x,y)从索引x处开始，删除y个 将删除的内容组成新数组返回
  • splice(x,y,...n)从索引x处开始，删除y个 将参数n及后面所有参数 插入到索引x前面 将删除的内容组成新数组返回
  • splice(x,0,..n) 将参数n以及后面所有参数 插入到索引x后面

3，返回值：截取后的值返回的值组成的新数组
4，原数组会发生改变

indexof

1，从左往右查找某个值在数组中的索引位置
2，参数：查找的字符
3，返回值：在数组中的索引值 如果查找的字符在数组中不存在就返回-1
4，不会改变原数组

    var arr = [1,2,3]
    var data0 = arr.indexOf(3)
    console.log(data0) //2

    // 如果查找的不存在 就返回-1
    var data = arr.indexOf(9)
    console.log(data) //-1

lastIndexof

1，作用：从右往左查找字符在数组中的索引位置
2，参数：查找的字符
3，返回值：在数组中的索引值 找不到返回-1
4，不会改变原数组

    var arr = [1,2,3,1]
    var data0 = arr.indexOf(1)
    var data2 = arr.lastIndexOf(1)
    console.log(data0) // 0
    console.log(data2) // 3

    var arr = [1,2,3,]
    var data0 = arr.indexOf(1)
    var data2 = arr.lastIndexOf(1)
    console.log(data0) // 0
    console.log(data2) // 0

slice

1，克隆，截取
2，参数：

  • slice():不传参数表示克隆，返回克隆后的新数组
  • slice(0):传参数0表示克隆，返回克隆后的新数组
  • slice(x):传参数x表示从x开始截取到最后 将截取的值 组成新的数组返回
  • slice(-x):传的参数为负数 表示截取 从右往左截取 截取x个 组成新的数组返回
  • slice(x,y)从索引x开始街取到索引y 但是不包括索引y 包前不包后
  • 如果起始索引大于结束索引 不会截取任何内容 直接返回一个新的空数组

3，返回值：截取，克隆后的新数组
4，不会改变原数组

sort()

1，作用：排序
2，参数：

  • arr.sort(function(a,b){a-b})
  • arr.sort(function(a,b){b-a})

3，返回值：排好序的数组
4，原数组会发生改变

    var arr = [3,14,1,'c','e',2,5,26,100]
    // 直接使用sort() 默认情况下 只能排序10以内的数字
    // arr.sort()
    // console.log(arr) //   [1, 100, 14, 2, 26, 3, 5]

        // 从小到大
    arr.sort(function(a,b){
        return a-b
    })
    console.log(arr) // [1, 2, 3, 5, 14, 26, 100]

    // 从大到小排序
    arr.sort(function(a,b){
        return b-a
    })

reverse()

1，排列 将原数组翻转过来
2，参数：无
3，返回值：排列好的数组
4，会改变原数组

    var arr3 = [1,2,3,4]
    arr3.reverse();
    console.log(arr3) // [4,3,2,1]

concat

1，作用：字符串拼接 默认展开一层数组
2，参数：任意数据类型的值
3，返回值：拼接后的新数组
4，不会改变原数组

     var arr1 = [1,2,3]
    var arr2 = ['a','b','c']
    var res1 = arr1.concat(arr2);
    console.log(res1) // [1, 2, 3, "a", "b", "c"]

    var res2 = arr2.concat(arr1);
    console.log(res2) // ["a", "b", "c", 1, 2, 3]

    // concat默认展开一层数组
    var arr3 = ['a','b','c',['A','B','C']]
    var res3 = arr1.concat(arr3);
    console.log(res3) // [1, 2, 3, "a", "b", "c", Array(3)]

toString()

1，作用：将数组转换成字符串
2，参数：无
3，返回值：转换后的字符串
4，不会改变原数组

    var arr = [1,2,3]
    var res = arr.toString(); 
    console.log(res) // '1,2,3'

join()

1，作用：将数组中的每项按照指定字符进行分割成字符串
2，参数：指定字符
3，返回值：分割后的新的字符串
4，不会改变原数组

    var a = ['+'];
    var arr2 = [1,2,3,'da'];
    var res2 = arr2.join(a)
    console.log(res2) // 1+2+3+da

eval()

将字符串当做JS代码来执行

相当于小型的JS解析器

    var arra = [1,2,3,4,5]
    var total = eval(arra.join('+'))// eval('1+2+3+4+5') => 15
    console.log(arra.join('+')) // '1+2+3+4+5'
    console.log(total) // 15
    
    // 相当于声明了一个变量
    eval('var num = 100')
