    var res = arr.map(function (item, index, input) {

   // input：原数组
   return item * 2
 })

forEach()和map()的区别：

1，返回值：forEach()的返回值是undefined；map()的返回值是一个新数组
2，forEach()对于空数组不会调用回调函数；map()不会对空数组进行检测，不会改变原数组

定时器

setTimeout:间隔一段时间只执行一次
setTnterval:每间隔一段时间 定时器就会执行一次

   var timer1 =  setTimeout(function(){
        console.log('一秒钟之后，会打印出这句话')
    },1000)
    
   var timer2 setInterval(function(){
        console.log('每隔两秒钟，就会打印出这句话')
    },2000)
    
    function timeout(){
        console.log('setTimeout只执行一次的定时器');
        console.log('setInterval每隔一段时间就会执行一次的定时器')
    }
    setTimeout(timeout,3000)
    
    // 清除定时器
    clearTimeout(timer1)
    clearInterval(timer2)
